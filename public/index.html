const res = await fetch(
  "https://x402trustlayer.xyz/api/guard/pre-x402",
  {
    method: "POST",
    headers: { "content-type": "application/json" },
    body: JSON.stringify({
      agentId: "buyer-1",
      walletAddress: wallet,
      targetUrl: downstreamUrl,
      estimatedCostUsdc: 0.05,
      policy: {
        dailyCapUsdc: 10,
        perCallCapUsdc: 0.5
      }
    })
  }
);

const gate = await res.json();
if (gate.allowed) {
  await x402Fetch(downstreamUrl);
}
